cmake_minimum_required(VERSION 2.8.3)
project(vigir_object_template_server)

## Find catkin and any catkin packages on which
## this package depends at build time
find_package(catkin REQUIRED COMPONENTS roscpp rospy tf tf_conversions pluginlib std_msgs geometry_msgs moveit_core moveit_ros_planning shape_msgs visualization_msgs vigir_grasp_msgs vigir_object_template_msgs nodelet cmake_modules)

find_package(TinyXML REQUIRED)

## Declare things to be passed to other projects
catkin_package(	
	INCLUDE_DIRS src
	LIBRARIES ${PROJECT_NAME} 
        CATKIN_DEPENDS roscpp rospy tf tf_conversions pluginlib std_msgs geometry_msgs moveit_core moveit_ros_planning shape_msgs visualization_msgs vigir_grasp_msgs vigir_object_template_msgs nodelet
        DEPENDS TinyXML
)

## Here we specify the list of source files, including the output of
## the previous command which is stored in ``${MOC_FILES}``.
set(SOURCE_FILES
  src/main.cpp
)

set(SOURCE_FILES_LIB
  src/object_template_server.cpp
)

include_directories(${catkin_INCLUDE_DIRS} ${EIGEN_INCLUDE_DIRS} src ${TinyXML_INCLUDE_DIRS})

add_library(${PROJECT_NAME} ${SOURCE_FILES_LIB})
target_link_libraries(${PROJECT_NAME} ${catkin_LIBRARIES} ${TinyXML_LIBRARIES})

add_dependencies(${PROJECT_NAME} ${catkin_EXPORTED_TARGETS})

add_executable(object_template_server ${SOURCE_FILES})
target_link_libraries(object_template_server ${catkin_LIBRARIES} ${PROJECT_NAME})

add_dependencies(object_template_server ${catkin_EXPORTED_TARGETS} vigir_object_template_msgs_generate_messages_cpp)

install(FILES
  vigir_object_template_server.xml
  DESTINATION ${CATKIN_PACKAGE_SHARE_DESTINATION})

install( TARGETS
    object_template_server
    DESTINATION ${CATKIN_PACKAGE_BIN_DESTINATION}
)
install(TARGETS ${PROJECT_NAME}
  LIBRARY DESTINATION ${CATKIN_PACKAGE_LIB_DESTINATION}
  )
